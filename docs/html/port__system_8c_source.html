61static void system_clock_config(void)
62{
64 /* Power controller (PWR) */
65 /* Control the main internal voltage regulator output voltage to achieve a trade-off between performance and power consumption when the device does not operate at the maximum frequency */
66 PWR->CR &= ~PWR_CR_VOS; // Clean and set value
67 PWR->CR |= (PWR_CR_VOS & (POWER_REGULATOR_VOLTAGE_SCALE3 << PWR_CR_VOS_Pos));
68
69 /* Initializes the RCC Oscillators. */
70 /* Adjusts the Internal High Speed oscillator (HSI) calibration value.*/
71 RCC->CR &= ~RCC_CR_HSITRIM; // Clean and set value
72 RCC->CR |= (RCC_CR_HSITRIM & (RCC_HSI_CALIBRATION_DEFAULT << RCC_CR_HSITRIM_Pos));
73
74 /* RCC Clock Config */
75 /* Initializes the CPU, AHB and APB buses clocks */
76 /* To correctly read data from FLASH memory, the number of wait states (LATENCY)
77 must be correctly programmed according to the frequency of the CPU clock
78 (HCLK) and the supply voltage of the device. */
79
80 /* Increasing the number of wait states because of higher CPU frequency */
81 FLASH->ACR = FLASH_ACR_LATENCY_2WS; /* Program the new number of wait states to the LATENCY bits in the FLASH_ACR register */
82
83 /* Change in clock source is performed in 16 clock cycles after writing to CFGR */
84 RCC->CFGR &= ~RCC_CFGR_SW; // Clean and set value
85 RCC->CFGR |= (RCC_CFGR_SW & (RCC_CFGR_SW_HSI << RCC_CFGR_SW_Pos));
86
87 /* Update the SystemCoreClock global variable */
88 SystemCoreClock = HSI_VALUE >> AHBPrescTable[(RCC->CFGR & RCC_CFGR_HPRE) >> RCC_CFGR_HPRE_Pos];
89
90 /* Configure the source of time base considering new system clocks settings */
91 SysTick_Config(SystemCoreClock / (1000U / TICK_FREQ_1KHZ)); /* Set Systick to 1 ms */
92}

94size_t port_system_init()
95{
96 /* Reset of all peripherals, Initializes the Flash interface and the Systick. */
97 /* Configure Flash prefetch, Instruction cache, Data cache */
98 /* Instruction cache enable */
99 FLASH->ACR |= FLASH_ACR_ICEN;
100
101 /* Data cache enable */
102 FLASH->ACR |= FLASH_ACR_DCEN;
103
104 /* Prefetch cache enable */
105 FLASH->ACR |= FLASH_ACR_PRFTEN;
106
107 /* Set Interrupt Group Priority */
108 NVIC_SetPriorityGrouping(NVIC_PRIORITY_GROUP_4);
109
110 /* Use systick as time base source and configure 1ms tick (default clock after Reset is HSI) */
111 /* Configure the SysTick IRQ priority. It must be the highest (lower number: 0)*/
112 NVIC_SetPriority(SysTick_IRQn, NVIC_EncodePriority(NVIC_GetPriorityGrouping(), 0U, 0U)); /* Tick interrupt priority */
113
114 /* Init the low level hardware */
115 /* Reset and clock control (RCC) */
116 RCC->APB2ENR |= RCC_APB2ENR_SYSCFGEN; /* Syscfg clock enabling */
117
118 /* Peripheral clock enable register */
119 RCC->APB1ENR |= RCC_APB1ENR_PWREN; /* PWREN: Power interface clock enable */
120
121 /* Configure the system clock */
122 system_clock_config();
123
124 return 0;
125}

200void port_system_gpio_config(GPIO_TypeDef *port, uint8_t pin, uint8_t mode, uint8_t pupd)
201{
202 /*Enables the clock of the GPIOX*/
203 if(port == GPIOA)
204 {
205 RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN;
206 }
207 else if(port == GPIOB)
208 {
209 RCC->AHB1ENR |= RCC_AHB1ENR_GPIOBEN;
210 }
211 else if(port == GPIOC)
212 {
213 RCC->AHB1ENR |= RCC_AHB1ENR_GPIOCEN;
214 }
215
216 /*Sets the PIN MODE*/
217 port->MODER &= ~BASE_MASK_TO_POS(BASE_MODER_MASK, 2*pin);
218 port->MODER |= BASE_MASK_TO_POS(mode, 2*pin);
219
220 /*Sets the pull-up/pull-down resistance*/
221 port->PUPDR &= ~BASE_MASK_TO_POS(BASE_PUPDR_MASK,2*pin);
222 port->PUPDR |= BASE_MASK_TO_POS(pupd,2*pin);
223}

252void port_system_gpio_config_exti(GPIO_TypeDef *port, uint8_t pin, uint32_t mode)
253{
254 /*Enables system controller*/
255 RCC->APB2ENR |= RCC_APB2ENR_SYSCFGEN;
256
257 /*Associate the exti with the GPIO*/
258 uint8_t mascara;
259 SYSCFG->EXTICR[pin/4] &= ~BASE_MASK_TO_POS(BASE_EXTIX_MASK,4*(pin%4));
260 if(port == GPIOA)
261 {
262 mascara = 0x00;
263 }
264 else if(port == GPIOB)
265 {
266 mascara = 0x01;
267 }
268 else if(port == GPIOC)
269 {
270 mascara = 0x02;
271 }
272 SYSCFG->EXTICR[pin/4] |= BASE_MASK_TO_POS(mascara, 4*(pin%4));
273
274 /*Interruption trigger*/
275 if((mode & BASE_MASK_TO_POS(0x3,0)) == TRIGGER_RISING_EDGE)
276 {
277 EXTI->RTSR |= BIT_POS_TO_MASK(pin); /*Rising edge*/
278 }
279 else if((mode & BASE_MASK_TO_POS(0x3,0)) == TRIGGER_FALLING_EDGE)
280 {
281 EXTI->FTSR |= BIT_POS_TO_MASK(pin); /*Falling edge*/
282 }
283 else if((mode & BASE_MASK_TO_POS(0x3,0)) == TRIGGER_BOTH_EDGE)
284 {
285 EXTI->RTSR |= BIT_POS_TO_MASK(pin); /*Rising edge*/
286 EXTI->FTSR |= BIT_POS_TO_MASK(pin); /*Falling edge*/
287 }
288
289 /*Enable local interrupt and event*/
290 if((mode & BASE_MASK_TO_POS(0x3,2)) == TRIGGER_ENABLE_INTERR_REQ){
291 EXTI->IMR |= BIT_POS_TO_MASK(pin);
292 }
293 else if((mode & BASE_MASK_TO_POS(0x3,2)) == TRIGGER_ENABLE_EVENT_REQ)
294 {
295 EXTI->EMR |= BIT_POS_TO_MASK(pin);
296 }
297 else if((mode & BASE_MASK_TO_POS(0x3,2)) == TRIGGER_ENABLE_EVENT_AND_INTERR_REQ)
298 {
299 EXTI->EMR |= BIT_POS_TO_MASK(pin);
300 EXTI->IMR |= BIT_POS_TO_MASK(pin);
301 }
302
303}
