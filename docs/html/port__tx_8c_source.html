42static void _timer_symbol_setup()
43{
44 /* Enable clock source */
45 RCC->APB2ENR |= RCC_APB2ENR_TIM1EN;
46
47 /* Disables counter register and enables preload register*/
48 TIM1->CR1 &= ~TIM_CR1_CEN;
49 TIM1->CR1 |= TIM_CR1_ARPE;
50
51 /* Reset TIM1 counter*/
52 TIM1->CNT = 0x0;
53
54 /*Set the prescaler*/
55 uint64_t psc = (SystemCoreClock/1000000*NEC_TX_TIMER_TICK_BASE_US)/((65535+1)) - 1;
56 TIM1->PSC = psc;
57
58 /*Set the ARR value*/
59 uint64_t arr = (SystemCoreClock*NEC_TX_TIMER_TICK_BASE_US)/(1000000*(TIM1->PSC+1)) - 1;
60 TIM1->ARR = arr;
61
62 /*Generate event update*/
63 TIM1->EGR = TIM_EGR_UG;
64
65 /*Cleanup interruptions*/
66 TIM1->SR = ~TIM_SR_UIF;
67
68 /*Enable interruptions*/
69 TIM1->DIER |= TIM_DIER_UIE;
70
71 /* Finally, set the priority and enable interruptions globally. */
72 NVIC_SetPriority(TIM1_UP_TIM10_IRQn, NVIC_EncodePriority(NVIC_GetPriorityGrouping(), 1, 0)); /* Priority 1, sub-priority 0 */
73 NVIC_EnableIRQ(TIM1_UP_TIM10_IRQn); /* Enable interrupt */
74}

81static void _timer_pwm_setup(uint32_t tx_id)
82{
83 /*Enable TIM2 clock*/
84 RCC->APB1ENR |= RCC_APB1ENR_TIM2EN;
85
86 /*Clock source (not necessary as it is default)*/
87 TIM2->CR1 = 0x0000;
88
89 /*Disable counter register and enable autoreload preload*/
90 TIM2->CR1 &= ~TIM_CR1_CEN;
91 TIM2->CR1 |= TIM_CR1_ARPE;
92
93 /*Reset the timer count*/
94 TIM2->CNT = 0;
95
96 /*Set the value of the prescaler*/
97 uint32_t psc = ceil(SystemCoreClock/NEC_PWM_FREQ_HZ/65536) - 1;
98 TIM2->PSC = psc;
99
100 /*Set the value of the autoreload register*/
101 uint64_t arr = (SystemCoreClock/NEC_PWM_FREQ_HZ)/(TIM2->PSC+1) - 1;
102 TIM2->ARR = arr; //Values for PWM period
103
104 /*Generate an update event*/
105 TIM2->EGR = TIM_EGR_UG;
106
107 /*Disable output commpare of CH3*/
108 TIM2->CCER &= ~TIM_CCER_CC3E;
109
110 /*Set both mode PWM 1 and enable preload register of CH3*/
111 TIM2->CCMR2 |= TIM_AS_PWM1_MASK;
112 TIM2->CCMR2 |= TIM_CCMR2_OC3PE;
113
114 /*Set the pulse width of CH3*/
115 uint32_t ccr3 = NEC_PWM_DC * (TIM2->ARR+1);
116 TIM2->CCR3 = ccr3;
117}
