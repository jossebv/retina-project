107void fsm_send_NEC_code(uint8_t tx_id, uint32_t code)
108{
109 uint32_t bit_mask = 0x80000000; /*Mask on bit 32*/
110 port_tx_symbol_tmr_start();
111
112 _send_NEC_burst(tx_id, NEC_TX_PROLOGUE_TICKS_ON, NEC_TX_PROLOGUE_TICKS_OFF);
113 do
114 {
115 if(code & bit_mask) /* Send a 1*/
116 {
117 _send_NEC_burst(tx_id, NEC_TX_SYM_1_TICKS_ON, NEC_TX_SYM_1_TICKS_OFF);
118 }
119 else /* Send a 0*/
120 {
121 _send_NEC_burst(tx_id, NEC_TX_SYM_0_TICKS_ON, NEC_TX_SYM_0_TICKS_OFF);
122 }
123 bit_mask >>= 1;
124 } while (bit_mask>0);
125 _send_NEC_burst(tx_id, NEC_TX_EPILOGUE_TICKS_ON, NEC_TX_EPILOGUE_TICKS_OFF); //Envio epilogo
126 port_tx_symbol_tmr_stop();
127}
