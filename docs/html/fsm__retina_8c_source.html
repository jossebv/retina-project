57void _process_rgb_code(uint8_t rgb_id, uint32_t code)
58{
59 if (code == MY_ON_BUTTON)
60 {
61 if (rgb_state)
62 {
63 port_rgb_set_color(rgb_id, LOW, LOW, LOW);
64 rgb_state = !rgb_state;
65 return;
66 }
67
68 if (color_mem == 0x0)
69 {
70 port_rgb_set_color(rgb_id, HIGH, HIGH, HIGH);
71 }
72 else
73 {
74 _process_rgb_code(rgb_id, color_mem);
75 }
76 rgb_state = !rgb_state;
77 return;
78 }
79 else if (code == MY_OFF_BUTTON)
80 {
81 port_rgb_set_color(rgb_id, LOW, LOW, LOW);
82 return;
83 }
84
85 if (code == MY_RED_BUTTON)
86 {
87 port_rgb_set_color(rgb_id, HIGH, LOW, LOW);
88 }
89 else if (code == MY_GREEN_BUTTON)
90 {
91 port_rgb_set_color(rgb_id, LOW, HIGH, LOW);
92 }
93 else if (code == MY_BLUE_BUTTON)
94 {
95 port_rgb_set_color(rgb_id, LOW, LOW, HIGH);
96 }
97 else if (code == LIL_WHITE_BUTTON)
98 {
99 port_rgb_set_color(rgb_id, HIGH, HIGH, HIGH);
100 }
101 color_mem = code;
102}
