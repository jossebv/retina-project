167void SystemCoreClockUpdate(void)
168{
169 uint32_t tmp = 0, pllvco = 0, pllp = 2, pllsource = 0, pllm = 2;
170
171 /* Get SYSCLK source -------------------------------------------------------*/
172 tmp = RCC->CFGR & RCC_CFGR_SWS;
173
174 switch (tmp)
175 {
176 case 0x00: /* HSI used as system clock source */
177 SystemCoreClock = HSI_VALUE;
178 break;
179 case 0x04: /* HSE used as system clock source */
180 SystemCoreClock = HSE_VALUE;
181 break;
182 case 0x08: /* PLL used as system clock source */
183
184 /* PLL_VCO = (HSE_VALUE or HSI_VALUE / PLL_M) * PLL_N
185 SYSCLK = PLL_VCO / PLL_P
186 */
187 pllsource = (RCC->PLLCFGR & RCC_PLLCFGR_PLLSRC) >> 22;
188 pllm = RCC->PLLCFGR & RCC_PLLCFGR_PLLM;
189
190 if (pllsource != 0)
191 {
192 /* HSE used as PLL clock source */
193 pllvco = (HSE_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
194 }
195 else
196 {
197 /* HSI used as PLL clock source */
198 pllvco = (HSI_VALUE / pllm) * ((RCC->PLLCFGR & RCC_PLLCFGR_PLLN) >> 6);
199 }
200
201 pllp = (((RCC->PLLCFGR & RCC_PLLCFGR_PLLP) >> 16) + 1) * 2;
202 SystemCoreClock = pllvco / pllp;
203 break;
204 default:
205 SystemCoreClock = HSI_VALUE;
206 break;
207 }
208 /* Compute HCLK frequency --------------------------------------------------*/
209 /* Get HCLK prescaler */
210 tmp = AHBPrescTable[((RCC->CFGR & RCC_CFGR_HPRE) >> 4)];
211 /* HCLK frequency */
212 SystemCoreClock >>= tmp;
213}

312void port_system_gpio_config(GPIO_TypeDef *port, uint8_t pin, uint8_t mode, uint8_t pupd)
313{
314 /*Enables the clock of the GPIOX*/
315 if (port == GPIOA)
316 {
317 RCC->AHB1ENR |= RCC_AHB1ENR_GPIOAEN;
318 }
319 else if (port == GPIOB)
320 {
321 RCC->AHB1ENR |= RCC_AHB1ENR_GPIOBEN;
322 }
323 else if (port == GPIOC)
324 {
325 RCC->AHB1ENR |= RCC_AHB1ENR_GPIOCEN;
326 }
327
328 /*Sets the PIN MODE*/
329 port->MODER &= ~BASE_MASK_TO_POS(BASE_MODER_MASK, 2 * pin);
330 port->MODER |= BASE_MASK_TO_POS(mode, 2 * pin);
331
332 /*Sets the pull-up/pull-down resistance*/
333 port->PUPDR &= ~BASE_MASK_TO_POS(BASE_PUPDR_MASK, 2 * pin);
334 port->PUPDR |= BASE_MASK_TO_POS(pupd, 2 * pin);

364void port_system_gpio_config_exti(GPIO_TypeDef *port, uint8_t pin, uint32_t mode)
365{
366 /*Enables system controller*/
367 RCC->APB2ENR |= RCC_APB2ENR_SYSCFGEN;
368
369 /*Associate the exti with the GPIO*/
370 uint8_t mascara;
371 SYSCFG->EXTICR[pin / 4] &= ~BASE_MASK_TO_POS(BASE_EXTIX_MASK, 4 * (pin % 4));
372 if (port == GPIOA)
373 {
374 mascara = 0x00;
375 }
376 else if (port == GPIOB)
377 {
378 mascara = 0x01;
379 }
380 else if (port == GPIOC)
381 {
382 mascara = 0x02;
383 }
384 SYSCFG->EXTICR[pin / 4] |= BASE_MASK_TO_POS(mascara, 4 * (pin % 4));
385
386 /*Interruption trigger*/
387 if ((mode & BASE_MASK_TO_POS(0x3, 0)) == TRIGGER_RISING_EDGE)
388 {
389 EXTI->RTSR |= BIT_POS_TO_MASK(pin); /*Rising edge*/
390 }
391 else if ((mode & BASE_MASK_TO_POS(0x3, 0)) == TRIGGER_FALLING_EDGE)
392 {
393 EXTI->FTSR |= BIT_POS_TO_MASK(pin); /*Falling edge*/
394 }
395 else if ((mode & BASE_MASK_TO_POS(0x3, 0)) == TRIGGER_BOTH_EDGE)
396 {
397 EXTI->RTSR |= BIT_POS_TO_MASK(pin); /*Rising edge*/
398 EXTI->FTSR |= BIT_POS_TO_MASK(pin); /*Falling edge*/
399 }
400
401 /*Enable local interrupt and event*/
402 if ((mode & BASE_MASK_TO_POS(0x3, 2)) == TRIGGER_ENABLE_INTERR_REQ)
403 {
404 EXTI->IMR |= BIT_POS_TO_MASK(pin);
405 }
406 else if ((mode & BASE_MASK_TO_POS(0x3, 2)) == TRIGGER_ENABLE_EVENT_REQ)
407 {
408 EXTI->EMR |= BIT_POS_TO_MASK(pin);
409 }
410 else if ((mode & BASE_MASK_TO_POS(0x3, 2)) == TRIGGER_ENABLE_EVENT_AND_INTERR_REQ)
411 {
412 EXTI->EMR |= BIT_POS_TO_MASK(pin);
413 EXTI->IMR |= BIT_POS_TO_MASK(pin);
414 }
415}
