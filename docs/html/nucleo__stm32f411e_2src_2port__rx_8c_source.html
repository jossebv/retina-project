56static void _store_edge_tick(uint8_t rx_id)

68 uint16_t tick = (uint16_t)(TIM3->CNT);
69 receivers_arr[rx_id].edge_ticks[edge_idx] = tick;

81void _timer_rx_setup()
82{
83 /*Enable clock source*/
84 RCC->APB1ENR |= RCC_APB1ENR_TIM3EN;
85
86 /*Disables counter register and enables preload register*/
87 TIM3->CR1 &= ~TIM_CR1_CEN;
